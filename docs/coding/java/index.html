    Priority Queue

    class Pair implements Comparable<Pair> {
    int a;
    int b;
    Pair(int a, int b) {
        this.a = a;
        this.b = b;
    }
    public int compareTo(Pair o) {
        if(this.a < o.a) return -1;
        else if(this.a == o.a) return 0;
        else return 1;
    }
}
Queue<Pair> pq = new PriorityQueue<>();
pq.offer(new Pair(1, 2));
pq.poll();
pq.size();
pq.isEmpty();
pq.peek();
